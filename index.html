🔗 ตั้งค่า Google Sheets

📁 ข้อมูลเก็บอยู่ที่ไหน?
ข้อมูลจะถูกบันทึกใน ไฟล์ Google Sheets ที่คุณเปิด Apps Script ไว้ โดยอัตโนมัติ สร้าง 2 Sheet ภายในไฟล์นั้น:
  • Attendance — ข้อมูลการเช็คเวลาทั้งหมด (คอลัมน์: ID, วันที่, ชื่อพนักงาน, เวลาIN, 9-11, 13-15, 15-17, เวลาOUT, สถานะ, บันทึกเมื่อ)
  • DayNotes — หมายเหตุวันหยุด/วันพิเศษ

⚠️ ทำไมข้อมูลไม่ขึ้น? ตรวจสอบ 3 จุด:
  1. Deploy ครั้งใหม่หลังวาง Script → ต้อง New Deployment ทุกครั้งที่แก้โค้ด (ไม่ใช่ Manage Deployments)
  2. ตอน Authorize ต้องกด Allow ให้ครบ (อาจมีหน้า "Google hasn't verified this app" → กด Advanced → Go to ...)
  3. Who has access ต้องตั้งเป็น Anyone (ไม่ใช่ Anyone with Google account)
  4. Sheet จะสร้างตัวเองอัตโนมัติเมื่อมีการบันทึกครั้งแรก — ถ้ายังไม่มีข้อมูลให้ลอง 🧪 ทดสอบ แล้วบันทึก 1 รายการ
⚡ Script เวอร์ชันใหม่ แก้ปัญหา CORS แล้ว — ส่งข้อมูลผ่าน GET request แทน POST ทำงานได้จากไฟล์ในเครื่องโดยตรง ไม่ต้องใช้ server
1
เปิด Google Sheets → Extensions → Apps Script → ลบโค้ดเดิมออกทั้งหมด
2
คัดลอก Script ด้านล่าง → วางลงไปทั้งหมด → กด Ctrl+S
3
Deploy → New Deployment
· Type: Web app · Execute as: Me · Who has access: Anyone
→ กด Deploy → กด Authorize access → เลือก Google Account → กด Allow
→ Copy URL ที่ได้ (ขึ้นต้นด้วย https://script.google.com/macros/s/...)
4
วาง URL ด้านล่าง → กด เชื่อมต่อ → กด 🧪 ทดสอบ ให้ขึ้น ✅ แล้วบันทึกข้อมูล → ไปดูใน Sheets จะมี Sheet ชื่อ Attendance ปรากฏขึ้น
📄 Apps Script (คัดลอกทั้งหมดไปวาง)
// ═══════════════════════════════════════════════════
//  ระบบเช็คเวลา — Google Apps Script v2 (CORS-free)
//  - ใช้ doGet เพียงอย่างเดียว (ไม่มี doPost)
//  - ส่งข้อมูลผ่าน URL parameter ไม่มี preflight
//  - ทำงานได้จากไฟล์ HTML ในเครื่องโดยตรง
//  - ข้อมูลเก็บใน Sheet ชื่อ "Attendance" และ "DayNotes"
//    ภายในไฟล์ Google Sheets ที่ Deploy Script นี้
// ═══════════════════════════════════════════════════
const SHEET = 'Attendance';
const NOTES = 'DayNotes';

function doGet(e) {
  const p = e.parameter;
  const action = p.action || '';

  if (action === 'ping')     return ok({ pong: true });
  if (action === 'getAll')   return ok({ rows: getRows() });
  if (action === 'getNotes') return ok({ notes: getNotes() });
  if (action === 'save')     return ok(saveEntry(p));
  if (action === 'saveNote') return ok(saveNote(p));
  if (action === 'delete')   return ok(deleteEntry(p.id));

  return ok({ status: 'ready' });
}

function saveEntry(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(SHEET);
  if (!sh) {
    sh = ss.insertSheet(SHEET);
    const h = ['ID','วันที่','ชื่อพนักงาน','เวลาIN',
               '9-11','13-15','15-17','เวลาOUT','สถานะ','บันทึกเมื่อ'];
    sh.appendRow(h);
    sh.getRange(1, 1, 1, h.length)
      .setFontWeight('bold')
      .setBackground('#1D9E75')
      .setFontColor('white');
  }
  // ลบข้อมูลเดิมของคนนี้วันนี้ก่อน
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (String(data[i][1]) === String(p.date) && String(data[i][2]) === String(p.name)) {
      sh.deleteRow(i + 1);
    }
  }
  sh.appendRow([
    p.id || Utilities.getUuid().substring(0, 8),
    p.date,
    p.name,
    p.timeIn  || '',
    p.slot1   || '',
    p.slot2   || '',
    p.slot3   || '',
    p.timeOut || '',
    p.status  || 'normal',
    Utilities.formatDate(new Date(), 'Asia/Bangkok', 'dd/MM/yyyy HH:mm')
  ]);
  return { success: true };
}

function saveNote(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(NOTES);
  if (!sh) {
    sh = ss.insertSheet(NOTES);
    sh.appendRow(['วันที่', 'ประเภท', 'หมายเหตุ']);
    sh.getRange(1, 1, 1, 3)
      .setFontWeight('bold')
      .setBackground('#7F77DD')
      .setFontColor('white');
  }
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (String(data[i][0]) === String(p.date)) sh.deleteRow(i + 1);
  }
  if (p.note || p.type) {
    sh.appendRow([p.date, p.type || '', p.note || '']);
  }
  return { success: true };
}

function deleteEntry(id) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET);
  if (!sh) return { success: false };
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (String(data[i][0]) === String(id)) {
      sh.deleteRow(i + 1);
      break;
    }
  }
  return { success: true };
}

function getRows() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  const hdrs = data[0].map(String);
  return data.slice(1).map(r => {
    const o = {};
    hdrs.forEach((h, i) => {
      const val = r[i];
      // แปลง Date object เป็น YYYY-MM-DD string
      if (val instanceof Date) {
        o[h] = Utilities.formatDate(val, 'Asia/Bangkok', 'yyyy-MM-dd');
      } else {
        o[h] = val === null || val === undefined ? '' : String(val);
      }
    });
    return o;
  });
}

function getNotes() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(NOTES);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  const hdrs = data[0].map(String);
  return data.slice(1).map(r => {
    const o = {};
    hdrs.forEach((h, i) => {
      const val = r[i];
      if (val instanceof Date) {
        o[h] = Utilities.formatDate(val, 'Asia/Bangkok', 'yyyy-MM-dd');
      } else {
        o[h] = val === null || val === undefined ? '' : String(val);
      }
    });
    return o;
  });
}

function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
