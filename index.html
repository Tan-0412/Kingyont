// ═══════════════════════════════════════════════════
//  ระบบเช็คเวลา — Google Apps Script v3
//  - ใช้ doGet เพียงอย่างเดียว (CORS-free)
//  - ดึง Record sheet → หาเวลาเข้า(ต่ำสุด)/ออก(สูงสุด) อัตโนมัติ
//  - เปลี่ยน Employees → Name sheet (มีคอลัมน์ ลำดับที่)
//  - ข้อมูลเก็บใน Sheet: Attendance, DayNotes, Name
// ═══════════════════════════════════════════════════
const SHEET = 'Attendance';
const NOTES = 'DayNotes';
const EMPS  = 'Name';  // เปลี่ยนจาก Employees → Name

function doGet(e) {
  const p = e.parameter;
  const action = p.action || '';

  if (action === 'ping')           return ok({ pong: true });
  if (action === 'getAll')         return ok({ rows: getRows() });
  if (action === 'getNotes')       return ok({ notes: getNotes() });
  if (action === 'getEmps')        return ok({ emps: getEmps() });
  if (action === 'getRecords')     return ok({ records: getRecords(p.date) });
  if (action === 'syncFromRecord') return ok(syncFromRecord(p.date));
  if (action === 'debugRecord')    return ok(debugRecord());
  if (action === 'save')           return ok(saveEntry(p));
  if (action === 'saveNote')       return ok(saveNote(p));
  if (action === 'saveEmp')        return ok(saveEmp(p));
  if (action === 'deleteEmp')      return ok(deleteEmp(p.name));
  if (action === 'delete')         return ok(deleteEntry(p.id));

  return ok({ status: 'ready' });
}

function saveEntry(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(SHEET);
  const h = ['ID','รหัสพนักงาน','วันที่','ชื่อพนักงาน','เวลาIN','9-11','13-15','15-17','เวลาOUT','สถานะ','บันทึกเมื่อ'];
  if (!sh) {
    sh = ss.insertSheet(SHEET);
    sh.appendRow(h);
    sh.getRange(1, 1, 1, h.length).setFontWeight('bold').setBackground('#1D9E75').setFontColor('white');
  } else if (sh.getLastRow() === 0) {
    sh.appendRow(h);
    sh.getRange(1, 1, 1, h.length).setFontWeight('bold').setBackground('#1D9E75').setFontColor('white');
  }
  const data = sh.getDataRange().getValues();
  const hdrs = data[0].map(String);
  const ciEmpId = hdrs.indexOf('รหัสพนักงาน');
  const ciDate  = hdrs.indexOf('วันที่');
  const empId   = String(p.empId || '').trim();
  const name    = String(p.name || p.name1 || '').trim();
  const date    = String(p.date || '').trim();

  // ลบแถวเดิมด้วย empId+date เท่านั้น (ไม่พึ่งชื่อ เพราะชื่ออาจต่างกัน)
  if (empId && date && ciEmpId >= 0 && ciDate >= 0) {
    for (let i = data.length - 1; i >= 1; i--) {
      const rowEmpId = String(data[i][ciEmpId] || '').trim();
      // คอลัมน์วันที่อาจเป็น Date object หรือ string → แปลงให้เป็น yyyy-MM-dd เสมอ
      const rawDate = data[i][ciDate];
      const rowDate = rawDate instanceof Date
        ? Utilities.formatDate(rawDate, 'Asia/Bangkok', 'yyyy-MM-dd')
        : String(rawDate || '').trim().substring(0, 10);
      if (rowEmpId === empId && rowDate === date) sh.deleteRow(i + 1);
    }
  }

  sh.appendRow([
    p.id || Utilities.getUuid().substring(0, 8),
    empId,
    date,
    name,
    p.timeIn  || '',
    p.slot1   || '',
    p.slot2   || '',
    p.slot3   || '',
    p.timeOut || '',
    p.status  || 'normal',
    Utilities.formatDate(new Date(), 'Asia/Bangkok', 'dd/MM/yyyy HH:mm')
  ]);
  return { success: true };
}

function saveNote(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(NOTES);
  if (!sh) {
    sh = ss.insertSheet(NOTES);
    sh.appendRow(['วันที่', 'ประเภท', 'หมายเหตุ']);
    sh.getRange(1, 1, 1, 3)
      .setFontWeight('bold')
      .setBackground('#7F77DD')
      .setFontColor('white');
  }
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (String(data[i][0]) === String(p.date)) sh.deleteRow(i + 1);
  }
  if (p.note || p.type) {
    sh.appendRow([p.date, p.type || '', p.note || '']);
  }
  return { success: true };
}

function deleteEntry(id) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET);
  if (!sh) return { success: false };
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (String(data[i][0]) === String(id)) {
      sh.deleteRow(i + 1);
      break;
    }
  }
  return { success: true };
}

function getRows() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  const hdrs = data[0].map(String);
  // คอลัมน์ที่เป็น "เวลา" ให้ format HH:mm, คอลัมน์ที่เป็น "วันที่" ให้ format yyyy-MM-dd
  const timeColumns = ['เวลาIN', 'เวลาOUT', 'IN', 'OUT', 'timeIn', 'timeOut'];
  const dateColumns = ['วันที่', 'date', 'Date', 'วันที่ทำงาน'];
  return data.slice(1).map(r => {
    const o = {};
    hdrs.forEach((h, i) => {
      const val = r[i];
      if (val instanceof Date) {
        const hTrim = h.trim();
        if (timeColumns.some(tc => tc === hTrim)) {
          // format เป็น HH:mm สำหรับคอลัมน์เวลา
          o[h] = Utilities.formatDate(val, 'Asia/Bangkok', 'HH:mm');
        } else if (dateColumns.some(dc => dc === hTrim)) {
          // format เป็น yyyy-MM-dd สำหรับคอลัมน์วันที่
          o[h] = Utilities.formatDate(val, 'Asia/Bangkok', 'yyyy-MM-dd');
        } else {
          // คอลัมน์อื่น (เช่น บันทึกเมื่อ) — ลอง detect ว่าเป็นเวลาหรือวันที่
          const formatted = Utilities.formatDate(val, 'Asia/Bangkok', 'yyyy-MM-dd HH:mm');
          o[h] = formatted;
        }
      } else {
        o[h] = val === null || val === undefined ? '' : String(val);
      }
    });
    return o;
  });
}

function getNotes() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(NOTES);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  const hdrs = data[0].map(String);
  return data.slice(1).map(r => {
    const o = {};
    hdrs.forEach((h, i) => {
      const val = r[i];
      if (val instanceof Date) {
        o[h] = Utilities.formatDate(val, 'Asia/Bangkok', 'yyyy-MM-dd');
      } else {
        o[h] = val === null || val === undefined ? '' : String(val);
      }
    });
    return o;
  });
}

function getEmps() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(EMPS); // 'Name'
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  if (data.length < 2) return [];
  const hdrs = data[0].map(h => String(h).trim());

  return data.slice(1).map(r => {
    const o = {};
    hdrs.forEach((h, i) => { o[h] = r[i] == null ? '' : String(r[i]); });
    return o;
  }).filter(r => {
    // กรองเฉพาะแถวที่มีรหัสพนักงาน
    const id = r['รหัสพนักงาน'] || r['empId'] || r['รหัส'] || '';
    return id.trim() !== '';
  });
}

// ดึงข้อมูลจาก Sheet "Record" ตามวันที่ที่ระบุ
// หลักการ: แต่ละแถวใน Record คือ timestamp ของการสแกนการ์ด
//   → แตกเวลาออกจากคอลัมน์ วันเวลา (DD/MM/YYYY HH:mm:ss)
//   → จัดกลุ่มตาม empId + วันที่
//   → เวลาน้อยที่สุด = timeIn (สีเหลือง), เวลาสูงสุด = timeOut (สีเขียว)
// คืนค่าเป็น array ของ { empId, date, timeIn, timeOut }
function getRecords(filterDate) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName('Record');
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  if (data.length < 2) return [];
  const hdrs = data[0].map(h => String(h).trim());

  // หา index ของคอลัมน์สำคัญ (รองรับชื่อหลายแบบ)
  const idxEmpId   = hdrs.findIndex(h => ['รหัสพนักงาน','empId','ID','รหัส'].includes(h));
  const idxDatetime= hdrs.findIndex(h => ['วันเวลา','datetime','Datetime','วันที่เวลา','timestamp'].includes(h));
  // ถ้าไม่มีคอลัมน์ วันเวลา ให้ลองหาคอลัมน์ วันที่ และ เวลา แยก
  const idxDate    = hdrs.findIndex(h => ['วันที่','date','Date','วันที่ทำงาน'].includes(h));
  const idxTime    = hdrs.findIndex(h => ['เวลา','time','Time','เวลาสแกน'].includes(h));

  // map: key = "empId|date" → { empId, date, times: [HH:mm, ...] }
  const grouped = {};

  data.slice(1).forEach(r => {
    const empId = idxEmpId >= 0 ? String(r[idxEmpId] || '').trim() : '';
    if (!empId) return;

    let dateStr = '';  // YYYY-MM-DD
    let timeStr = '';  // HH:mm

    if (idxDatetime >= 0) {
      // ── โหมด 1: คอลัมน์เดียว "วันเวลา" เช่น "23/03/2026 9:34:01" ──
      const raw = r[idxDatetime];
      if (raw instanceof Date) {
        dateStr = Utilities.formatDate(raw, 'Asia/Bangkok', 'yyyy-MM-dd');
        timeStr = Utilities.formatDate(raw, 'Asia/Bangkok', 'HH:mm');
      } else {
        const s = String(raw || '').trim();
        // รูปแบบ DD/MM/YYYY HH:mm[:ss]
        const m = s.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{4})\s+(\d{1,2}):(\d{2})/);
        if (m) {
          dateStr = m[3] + '-' + m[2].padStart(2,'0') + '-' + m[1].padStart(2,'0');
          timeStr = m[4].padStart(2,'0') + ':' + m[5];
        } else {
          // รูปแบบ YYYY-MM-DD HH:mm[:ss]
          const m2 = s.match(/^(\d{4})[\/\-](\d{2})[\/\-](\d{2})\s+(\d{1,2}):(\d{2})/);
          if (m2) {
            dateStr = m2[1] + '-' + m2[2] + '-' + m2[3];
            timeStr = m2[4].padStart(2,'0') + ':' + m2[5];
          }
        }
      }
    } else if (idxDate >= 0) {
      // ── โหมด 2: คอลัมน์วันที่ + คอลัมน์เวลา แยกกัน ──
      const rawD = r[idxDate];
      if (rawD instanceof Date) {
        dateStr = Utilities.formatDate(rawD, 'Asia/Bangkok', 'yyyy-MM-dd');
      } else {
        const s = String(rawD || '').trim();
        const m = s.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{4})$/);
        if (m) dateStr = m[3] + '-' + m[2].padStart(2,'0') + '-' + m[1].padStart(2,'0');
        else if (/^\d{4}-\d{2}-\d{2}/.test(s)) dateStr = s.substring(0,10);
      }
      if (idxTime >= 0) {
        const rawT = r[idxTime];
        if (rawT instanceof Date) {
          timeStr = Utilities.formatDate(rawT, 'Asia/Bangkok', 'HH:mm');
        } else {
          const s = String(rawT || '').trim();
          const m = s.match(/^(\d{1,2}):(\d{2})/);
          if (m) timeStr = m[1].padStart(2,'0') + ':' + m[2];
        }
      }
    }

    if (!dateStr || !timeStr) return;
    if (filterDate && dateStr !== filterDate) return;

    const key = empId + '|' + dateStr;
    if (!grouped[key]) grouped[key] = { empId, date: dateStr, times: [] };
    grouped[key].times.push(timeStr);
  });

  // แปลง grouped → results พร้อม min/max
  return Object.values(grouped).map(g => {
    const sorted = g.times.slice().sort(); // HH:mm sort alphabetically = chronologically
    return {
      empId:   g.empId,
      date:    g.date,
      timeIn:  sorted[0],                              // เวลาน้อยสุด = IN
      timeOut: sorted.length > 1 ? sorted[sorted.length - 1] : ''  // มีแค่ครั้งเดียว → OUT ว่าง
    };
  });
}

// ════════════════════════════════════════════════════════
//  syncFromRecord(date)
//  - ใช้ Name sheet เป็น master list (11 คน)
//  - ดึงเวลา IN/OUT จาก Record sheet (min/max)
//  - ถ้าไม่มีใน Record → สร้างรายการว่างไว้ (ขาด/ลา)
//  - empId ที่อยู่ใน Record แต่ไม่อยู่ใน Name → ข้ามทิ้ง
// ════════════════════════════════════════════════════════
function syncFromRecord(filterDate) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  // 1. โหลด Name sheet → สร้าง masterList [{empId, name, order}]
  const nameSh = ss.getSheetByName(EMPS);
  if (!nameSh) return { saved: 0, skipped: 0, rows: [], error: 'ไม่พบ Sheet "' + EMPS + '"' };
  const nameData = nameSh.getDataRange().getValues();
  if (nameData.length < 2) return { saved: 0, skipped: 0, rows: [], error: 'Name sheet ว่าง' };
  const nh = nameData[0].map(h => String(h).trim());
  const niId   = nh.findIndex(h => ['รหัสพนักงาน','empId','รหัส'].includes(h));
  const niName = nh.findIndex(h => ['ชื่อ1','ชื่อ'].includes(h));
  const niOrd  = nh.findIndex(h => ['ลำดับที่','ลำดับ'].includes(h));
  if (niId < 0 || niName < 0) return { saved: 0, skipped: 0, rows: [], error: 'ไม่พบคอลัมน์ รหัสพนักงาน หรือ ชื่อ1 ใน Name sheet' };

  const masterList = nameData.slice(1)
    .map(r => ({
      empId: String(r[niId]  || '').trim(),
      name:  String(r[niName]|| '').trim(),
      order: niOrd >= 0 ? Number(r[niOrd] || 9999) : 9999
    }))
    .filter(e => e.empId && e.name)
    .sort((a, b) => a.order - b.order);

  if (masterList.length === 0) return { saved: 0, skipped: 0, rows: [], error: 'ไม่มีพนักงานใน Name sheet' };

  // 2. โหลด Record sheet → สร้าง recordMap {empId: {timeIn, timeOut}}
  const recordMap = {};
  const recList = getRecords(filterDate); // ใช้ function เดิมที่หา min/max แล้ว
  recList.forEach(r => { recordMap[r.empId] = { timeIn: r.timeIn, timeOut: r.timeOut }; });

  // 3. เตรียม Attendance sheet + header
  const attH = ['ID','รหัสพนักงาน','วันที่','ชื่อพนักงาน','เวลาIN','9-11','13-15','15-17','เวลาOUT','สถานะ','บันทึกเมื่อ'];
  let sh = ss.getSheetByName(SHEET);
  if (!sh) {
    sh = ss.insertSheet(SHEET);
    sh.appendRow(attH);
    sh.getRange(1, 1, 1, attH.length).setFontWeight('bold').setBackground('#1D9E75').setFontColor('white');
  } else if (sh.getLastRow() === 0) {
    sh.appendRow(attH);
    sh.getRange(1, 1, 1, attH.length).setFontWeight('bold').setBackground('#1D9E75').setFontColor('white');
  }

  // 4. ดึง key ที่มีอยู่ใน Attendance แล้ว เช็คด้วย รหัสพนักงาน|วันที่ (แม่นยำกว่าใช้ชื่อ)
  const attData = sh.getDataRange().getValues();
  const attHdrs = attData[0].map(String);
  const aiEmpId = attHdrs.indexOf('รหัสพนักงาน');
  const aiDate  = attHdrs.indexOf('วันที่');
  const aiName  = attHdrs.indexOf('ชื่อพนักงาน');
  const existingKeys = new Set();
  attData.slice(1).forEach(r => {
    const empId = aiEmpId >= 0 ? String(r[aiEmpId] || '').trim() : '';
    const rawDate = r[aiDate];
    const d = rawDate instanceof Date
      ? Utilities.formatDate(rawDate, 'Asia/Bangkok', 'yyyy-MM-dd')
      : String(rawDate || '').trim().substring(0, 10);
    const n = String(r[aiName] || '').trim();
    if (empId && d) existingKeys.add(empId + '|' + d);
    if (n && d)     existingKeys.add('name|' + n + '|' + d);
  });

  // 5. วน masterList → เขียน Attendance ทีละคน
  const now = Utilities.formatDate(new Date(), 'Asia/Bangkok', 'dd/MM/yyyy HH:mm');
  let saved = 0, skipped = 0;
  const savedRows = [];

  masterList.forEach(emp => {
    // เช็ค duplicate: empId|date หรือ ชื่อ|date (กันซ้ำทั้งสองแบบ)
    const keyById   = emp.empId + '|' + filterDate;
    const keyByName = 'name|' + emp.name + '|' + filterDate;
    if (existingKeys.has(keyById) || existingKeys.has(keyByName)) { skipped++; return; }

    const rec = recordMap[emp.empId];
    const newId = filterDate.replace(/-/g,'') + '_' + emp.empId;

    // ตรวจสอบสถานะสาย: ถ้า timeIn > 09:00 → สาย
    const LATE_THRESHOLD = '09:00';
    let status = 'normal';
    if (rec && rec.timeIn) {
      status = rec.timeIn > LATE_THRESHOLD ? 'late' : 'normal';
    }

    sh.appendRow([
      newId,
      emp.empId,
      filterDate,
      emp.name,
      rec ? (rec.timeIn  || '') : '',
      'อยู่', 'อยู่', 'อยู่',
      rec ? (rec.timeOut || '') : '',
      status,
      now
    ]);
    existingKeys.add(keyById);
    existingKeys.add(keyByName);
    saved++;
    savedRows.push({
      empId:   emp.empId,
      name:    emp.name,
      date:    filterDate,
      timeIn:  rec ? rec.timeIn  : '',
      timeOut: rec ? rec.timeOut : '',
      hasRecord: !!rec
    });
  });

  return { saved, skipped, rows: savedRows };
}

// debug: ดูข้อมูลดิบจาก Record sheet เพื่อวิเคราะห์ปัญหา
function debugRecord() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName('Record');
  if (!sh) return { error: 'ไม่พบ Sheet ชื่อ "Record"' };
  const data = sh.getDataRange().getValues();
  if (data.length < 2) return { error: 'Record sheet ว่าง (ไม่มีข้อมูล)', rows: data.length };

  const hdrs = data[0].map(h => String(h).trim());

  // ดูแถว 1-3 พร้อม type ของแต่ละ cell
  const samples = data.slice(1, 4).map(r => {
    const obj = {};
    hdrs.forEach((h, i) => {
      const val = r[i];
      const t = val instanceof Date ? 'Date' : typeof val;
      const fmtVal = val instanceof Date
        ? Utilities.formatDate(val, 'Asia/Bangkok', 'yyyy-MM-dd HH:mm:ss')
        : String(val ?? '');
      obj[h] = { raw: fmtVal, type: t };
    });
    return obj;
  });

  return {
    sheetName: 'Record',
    totalRows: data.length - 1,
    headers: hdrs,
    samples,
    // ทดสอบ getRecords โดยไม่ filter
    recordsParsed: getRecords(null).slice(0, 5)
  };
}

function saveEmp(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(EMPS);
  if (!sh) {
    sh = ss.insertSheet(EMPS);
    // ใช้ header ตาม Name sheet จริง
    const h = ['ชื่อนามสกุล','ชื่อ','ชื่อ1','ชื่อเล่น','รหัสพนักงาน','บริษัท','ฝ่าย','ทีม','ตำแหน่ง','ลำดับที่'];
    sh.appendRow(h);
    sh.getRange(1, 1, 1, h.length)
      .setFontWeight('bold')
      .setBackground('#E24B4A')
      .setFontColor('white');
  }
  const data = sh.getDataRange().getValues();
  const hdrs = data[0].map(String);
  // หา index ของ รหัสพนักงาน
  const idxId = hdrs.findIndex(h => ['รหัสพนักงาน','empId','รหัส'].includes(h));
  if (idxId < 0) return { success: false, error: 'ไม่พบคอลัมน์รหัสพนักงาน' };

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][idxId]) === String(p.empId || p.name)) {
      // อัปเดตคอลัมน์ที่มี
      const idxPos = hdrs.findIndex(h => ['ตำแหน่ง','pos'].includes(h));
      const idxOrd = hdrs.findIndex(h => ['ลำดับที่','ลำดับ','order'].includes(h));
      if (idxPos >= 0) sh.getRange(i+1, idxPos+1).setValue(p.pos || 'พนักงาน');
      if (idxOrd >= 0 && p.order) sh.getRange(i+1, idxOrd+1).setValue(Number(p.order));
      return { success: true };
    }
  }
  // เพิ่มแถวใหม่ (กรณีที่ saveEmp ถูกเรียกจาก frontend เพื่อเพิ่มพนักงาน)
  const newRow = new Array(hdrs.length).fill('');
  hdrs.forEach((h, i) => {
    if (['รหัสพนักงาน','empId'].includes(h)) newRow[i] = p.empId || '';
    if (['ชื่อ1','ชื่อ'].includes(h)) newRow[i] = p.name || '';
    if (['ตำแหน่ง','pos'].includes(h)) newRow[i] = p.pos || 'พนักงาน';
    if (['ลำดับที่','ลำดับ'].includes(h)) newRow[i] = Number(p.order) || data.length;
  });
  sh.appendRow(newRow);
  return { success: true };
}

function deleteEmp(name) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(EMPS);
  if (!sh) return { success: false };
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (String(data[i][0]) === String(name)) { sh.deleteRow(i+1); return { success: true }; }
  }
  return { success: false };
}
