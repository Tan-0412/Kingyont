📄 Apps Script
const SHEET_NAME = 'Attendance';
const NOTE_SHEET = 'DayNotes';

function doGet(e) {
  const a = e.parameter.action;
  if (a === 'getAll')    return respond({rows: getRows()});
  if (a === 'getNotes')  return respond({notes: getNotes()});
  return respond({error: 'unknown action'});
}

function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    if (d.action === 'saveNote') return saveNote(d);
    return saveEntry(d);
  } catch(err) {
    return respond({error: err.toString()});
  }
}

function saveEntry(d) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) {
    sh = ss.insertSheet(SHEET_NAME);
    const hdrs = ['ID','วันที่','ชื่อพนักงาน','เวลาIN','9-11','13-15','15-17','เวลาOUT','สถานะ','บันทึกเมื่อ'];
    sh.appendRow(hdrs);
    sh.getRange(1,1,1,hdrs.length).setFontWeight('bold')
      .setBackground('#1D9E75').setFontColor('white');
  }
  // Remove existing same person+date
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (data[i][1] == d.date && data[i][2] == d.name) sh.deleteRow(i + 1);
  }
  sh.appendRow([d.id||Utilities.getUuid().substring(0,8),
    d.date, d.name, d.timeIn, d.slot1, d.slot2, d.slot3,
    d.timeOut, d.status, new Date().toLocaleString('th-TH')]);
  return respond({success: true});
}

function saveNote(d) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(NOTE_SHEET);
  if (!sh) {
    sh = ss.insertSheet(NOTE_SHEET);
    sh.appendRow(['วันที่','ประเภท','หมายเหตุ']);
    sh.getRange(1,1,1,3).setFontWeight('bold').setBackground('#7F77DD').setFontColor('white');
  }
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (data[i][0] == d.date) sh.deleteRow(i + 1);
  }
  if (d.note) sh.appendRow([d.date, d.type, d.note]);
  return respond({success: true});
}

function getRows() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  const hdrs = data[0];
  return data.slice(1).map(r => {
    const obj = {};
    hdrs.forEach((h,i) => obj[String(h)] = r[i]);
    return obj;
  });
}

function getNotes() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(NOTE_SHEET);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  const hdrs = data[0];
  return data.slice(1).map(r => {
    const obj = {};
    hdrs.forEach((h,i) => obj[String(h)] = r[i]);
    return obj;
  });
}

function respond(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
